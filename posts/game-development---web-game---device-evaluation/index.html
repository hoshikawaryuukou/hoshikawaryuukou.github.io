Console

(function () {
    let lastTime = performance.now();
    let frame = 0;

    function loop() {
        let now = performance.now();
        frame++;
        if (now > lastTime + 1000) {
            console.log(`FPS: ${frame}`);
            lastTime = now;
            frame = 0;
        }
        requestAnimationFrame(loop);
    }
    loop();
})();
